 case class Kleisli[M[_], A, B](run: A => M[B]) {
     def andThen[C](k: Kleisli[M, B, C])(implicit b: Monad[M]) =
       Kleisli[M, A, C]((a: A) => b.bind(this(a))(k.run))

    def compose[C](k: Kleisli[M, C, A])(implicit b: Monad[M]) =
      k andThen this
 }
